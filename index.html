1、检索记录

1.1从表中检索所有的行和列

显示的指定哪些列，可以清楚的查询返回哪些列

select * from emp；--不推荐
select empno， empname，job，sal，comn from emp；--推荐

1.2从表中检索部分行

select * from emp where deptno = 10；

1.3查找满足多个条件的行

查询部门10中所有的员工，所有得提成的员工，以及部门20中工资不超过2000美金的员工 。 

select * from emp where deptno = 10 
or comn is not null
or sal < 2000 and deptno = 20;


select * from emp where (deptno = 10 or comm is not null or sal < 2000 ) 
and deptno = 20;

1.4、从表中检索部分列

select ename，deptno, sal from emp;--【查找员工姓名，部门编号，奖金三列的员工信息】

1.5、为列取有意义的名称

select sal as salary ,comm as commission from emp;

1.6在where子句中引用取别名的列

将查询作为内联视图就可以引用其中区别名的列

select sal as salary ,comm as commission from emp where salary < 5000;--错误

select *from (
    select sal as salary ,comm as commission from emp ) x where salary < 5000;--正确

1.7 连接列值

select ename,job from emp where deptno = 10 ;

select ename + ' work as A ' + job as mrg from emp where deptno = 10 ;--Sql server
select concat(ename, 'work as A', job) as mrg from emp where deptno = 10 ;--Mysql 和SqlServer  Oracle
--select ename || ' work as A ' || job as mrg from emp where deptno = 10 ;--DB2 Oracle PostgreSQl

1.8 在select语句中使用条件逻辑 

--资金小于2000 输出underpaid， 大于2000就输出 overpaid ，其他输出 ok
select ename, sal,
    case when sal < 2000 then 'underpaid'
         when sal > 2000 then 'overpaid'
         else 'ok'
    end as status
from emp;
